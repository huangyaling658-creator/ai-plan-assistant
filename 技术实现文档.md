### 2.5 计划渲染引擎

将 AI 生成的自然文本渲染为可交互的计划面板。

#### 解析算法

```javascript
function parsePlanText(planText, completionMarker) {
    const lines = planText.split('\n');
    const sections = [];
    let currentSection = { title: '', items: [] };

    for (const line of lines) {
        const trimmed = line.trim();
        if (!trimmed) continue;

        // 识别分组标题
        if (trimmed.startsWith('## ') || trimmed.startsWith('### ')) {
            if (currentSection.title || currentSection.items.length) {
                sections.push(currentSection);
            }
            currentSection = { title: trimmed.replace(/^#+\s*/, ''), items: [] };
        }
        // 识别任务项
        else if (trimmed.match(/^[-*]\s/)) {
            currentSection.items.push({
                id: generateId(),
                text: trimmed.replace(/^[-*]\s/, ''),
                completed: isMarkedComplete(trimmed, completionMarker),
                raw: line
            });
        }
        // 其他行作为纯文本
        else {
            currentSection.items.push({
                id: generateId(),
                text: trimmed,
                completed: false,
                isPlainText: true,
                raw: line
            });
        }
    }
    if (currentSection.title || currentSection.items.length) {
        sections.push(currentSection);
    }
    return sections;
}
```

#### 完成状态识别（根据用户风格）

```javascript
function isMarkedComplete(line, completionMarker) {
    if (!completionMarker) return false;

    const markerMap = {
        'checkbox': /\[x\]/i,
        'emoji_check': /[✅☑✓]/,
        'emoji_cross': /[❌✗✘]/,
        'strikethrough': /~~.+~~/,
        'text_done': /已完成|done|完成/i,
        'default': /[✅☑✓\[x\]]/i
    };

    const pattern = markerMap[completionMarker] || markerMap['default'];
    return pattern.test(line);
}
```

#### 修改差异检测（Diff 算法）

```javascript
function detectModifications(original, modified) {
    const origLines = original.split('\n').filter(l => l.trim());
    const modLines = modified.split('\n').filter(l => l.trim());

    const origSet = new Set(origLines);
    const modSet = new Set(modLines);

    return {
        added: modLines.filter(l => !origSet.has(l)),
        removed: origLines.filter(l => !modSet.has(l)),
        unchanged: origLines.filter(l => modSet.has(l)),
        original: original,
        modified: modified,
        timestamp: Date.now(),
        changeRatio: 1 - (origLines.filter(l => modSet.has(l)).length /
                         Math.max(origLines.length, 1))
    };
}
```

## 三、AI 模型层

### 3.1 OpenRouter API 封装

#### 核心调用函数

```javascript
async function callOpenRouter(messages, options = {}) {
    const config = loadData('planai_api_config') || {};
    const apiKey = options.apiKey || config.apiKey;
    const model = options.model || config.model || 'google/gemini-3-flash-preview';

    if (!apiKey) {
        throw new Error('请先在设置中配置 OpenRouter API Key');
    }

    const response = await fetch('https://openrouter.ai/api/v1/chat/completions', {
        method: 'POST',
        headers: {
            'Authorization': `Bearer ${apiKey}`,
            'Content-Type': 'application/json',
            'HTTP-Referer': window.location.href,
            'X-Title': 'AI Plan Assistant'
        },
        body: JSON.stringify({
            model: model,
            messages: messages,
            temperature: options.temperature || 0.7,
            max_tokens: options.maxTokens || 4096
        })
    });

    if (!response.ok) {
        const err = await response.json().catch(() => ({}));
        throw new Error(err.error?.message || `API 请求失败 (${response.status})`);
    }

    const data = await response.json();
    return data.choices[0].message.content;
}
```

#### 错误处理与重试

```javascript
async function callWithRetry(messages, options = {}, maxRetries = 2) {
    for (let attempt = 0; attempt <= maxRetries; attempt++) {
        try {
            return await callOpenRouter(messages, options);
        } catch (error) {
            if (attempt === maxRetries) throw error;

            // 429 限流：等待后重试
            if (error.message.includes('429')) {
                await sleep(2000 * (attempt + 1));
                continue;
            }
            // 其他错误直接抛出
            throw error;
        }
    }
}
```

```javascript
function parseResponse(layer, rawResponse) {
    switch (layer) {
        case 'style_analysis':

            return extractJSON(rawResponse);

        case 'plan_generation':
        case 'tomorrow_plan':

        case 'daily_summary':
            // 返回 Markdown 格式的总结
            return rawResponse.trim();
    }
}

function extractJSON(text) {
    // 尝试直接解析
    try { return JSON.parse(text); } catch {}

    // 尝试从 ```json ... ``` 代码块中提取
    const codeBlockMatch = text.match(/```(?:json)?\s*([\s\S]*?)```/);
    if (codeBlockMatch) {
        try { return JSON.parse(codeBlockMatch[1].trim()); } catch {}
    }

    // 尝试找到第一个 { 和最后一个 } 之间的内容
    const braceMatch = text.match(/\{[\s\S]*\}/);
    if (braceMatch) {
        try { return JSON.parse(braceMatch[0]); } catch {}
    }

    throw new Error('无法从 AI 响应中提取有效的 JSON');
}
```

### 5.1 风格画像合并算法

当用户上传多份历史计划时，需要将新提取的风格与已有画像合并：

```javascript
function mergeStyleProfiles(existing, newProfile) {
    if (!existing) return newProfile;

    const merged = { ...existing };

    // 字符串字段：新值优先（如果新的置信度更高）
    const stringFields = ['format', 'granularity', 'time_preference',
                         'language_style', 'weekend_pattern', 'completion_markers'];
    for (const field of stringFields) {
        if (newProfile[field] && newProfile.confidence !== 'low') {
            merged[field] = newProfile[field];
        }
    }

    // 数组字段：去重合并
    const arrayFields = ['categories', 'recurring_items', 'sample_expressions'];
    for (const field of arrayFields) {
        if (newProfile[field]) {
            const combined = [...(existing[field] || []), ...newProfile[field]];
            merged[field] = [...new Set(combined)];
        }
    }

    // 数值/范围字段：取平均或扩大范围
    if (newProfile.typical_daily_count) {
        merged.typical_daily_count = newProfile.typical_daily_count;
    }
    if (newProfile.peak_hours) {
        merged.peak_hours = newProfile.peak_hours;
    }

    // 布尔字段：新值覆盖
    const boolFields = ['buffer_habit', 'goal_linking'];
    for (const field of boolFields) {
        if (newProfile[field] !== undefined) {
            merged[field] = newProfile[field];
        }
    }

    // 置信度：取更高的
    const confLevels = { low: 1, medium: 2, high: 3 };
    merged.confidence = (confLevels[newProfile.confidence] || 0) >
                        (confLevels[existing.confidence] || 0)
                        ? newProfile.confidence : existing.confidence;

    // 更新时间戳
    merged._lastUpdated = Date.now();
    merged._updateCount = (existing._updateCount || 1) + 1;

    return merged;
}
```

```javascript
function applyProfileUpdates(currentProfile, updates) {
    if (!updates || !updates.style_profile_updates) return currentProfile;

    const updatedProfile = { ...currentProfile };
    const suggestions = updates.style_profile_updates;

    for (const [field, newValue] of Object.entries(suggestions)) {
        if (field in updatedProfile) {
            // 数组字段：合并
            if (Array.isArray(updatedProfile[field]) && Array.isArray(newValue)) {
                updatedProfile[field] = [...new Set([
                    ...updatedProfile[field], ...newValue
                ])];
            }
            // 其他字段：直接覆盖
            else {
                updatedProfile[field] = newValue;
            }
        }
    }

    updatedProfile._lastUpdated = Date.now();
    return updatedProfile;
}
```

### 5.4 完成率计算与趋势分析

```javascript
function calculateCompletionStats(planSections) {
    let total = 0, completed = 0;

    planSections.forEach(section => {
        section.items.forEach(item => {
            if (!item.isPlainText) {
                total++;
                if (item.completed) completed++;
            }
        });
    });

    return {
        total,
        completed,
        rate: total > 0 ? Math.round((completed / total) * 100) : 0
    };
}

function getCompletionTrend(days = 7) {
    const dailyPlans = loadData('planai_daily_plans') || {};
    const today = new Date();
    const trend = [];

    for (let i = days - 1; i >= 0; i--) {
        const date = new Date(today);
        date.setDate(date.getDate() - i);
        const key = formatDate(date);

        if (dailyPlans[key]) {
            const sections = parsePlanText(dailyPlans[key].content);
            const stats = calculateCompletionStats(sections);
            trend.push({ date: key, ...stats });
        }
    }

    return {
        data: trend,
        averageRate: trend.length > 0
            ? Math.round(trend.reduce((s, d) => s + d.rate, 0) / trend.length)
            : null,
        trend: describeTrend(trend)
    };
}

function describeTrend(data) {
    if (data.length < 3) return '数据不足';
    const recent = data.slice(-3).map(d => d.rate);
    const avg = recent.reduce((s, r) => s + r, 0) / recent.length;
    const first = recent[0];
    if (avg > first + 5) return '持续上升';
    if (avg < first - 5) return '持续下降';
    return '基本稳定';
}
```

```javascript
const STORAGE_PREFIX = 'planai_';

function saveData(key, value) {
    try {
        localStorage.setItem(key, JSON.stringify(value));
        return true;
    } catch (e) {
        if (e.name === 'QuotaExceededError') {
            console.warn('localStorage 空间不足，尝试清理旧数据');
            cleanOldData();
            localStorage.setItem(key, JSON.stringify(value));
        }
        return false;
    }
}

function loadData(key) {
    try {
        const raw = localStorage.getItem(key);
        return raw ? JSON.parse(raw) : null;
    } catch {
        return null;
    }
}

function cleanOldData() {
    // 清理 60 天前的日计划和总结
    const cutoff = Date.now() - 60 * 24 * 60 * 60 * 1000;
    ['planai_daily_plans', 'planai_summaries'].forEach(key => {
        const data = loadData(key) || {};
        for (const date in data) {
            if (data[date].timestamp < cutoff) delete data[date];
        }
        saveData(key, data);
    });
}
```

```javascript
const SUPABASE_URL = '';  // 用户配置

    supabase = window.supabase.createClient(url, anonKey);

// 统一数据访问层：自动选择 localStorage 或 Supabase
const DataLayer = {
    async save(table, data) {
        if (supabase) {
            const { error } = await supabase.from(table).upsert(data);
            if (error) throw error;
        } else {
            saveData(`planai_${table}`, data);
        }
    },

    async load(table, filters = {}) {
        if (supabase) {
            let query = supabase.from(table).select('*');
            for (const [key, value] of Object.entries(filters)) {
                query = query.eq(key, value);
            }
            const { data, error } = await query;
            if (error) throw error;
            return data;
        } else {
            return loadData(`planai_${table}`);
        }
    }
};
```

```javascript
// 防抖工具函数
function debounce(fn, ms) {
    let timer;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => fn(...args), ms);
    };
}

// API 并发锁
let _apiLock = false;
async function withApiLock(fn) {
    if (_apiLock) {
        throw new Error('正在处理上一个请求，请稍候');
    }
    _apiLock = true;
    try {
        return await fn();
    } finally {
        _apiLock = false;
    }
}
```

```javascript
// 日期格式化
function formatDate(date) {
    return date.toISOString().split('T')[0];  // '2026-02-28'
}

function getDayOfWeek() {
    return ['日', '一', '二', '三', '四', '五', '六'][new Date().getDay()];
}

function getCurrentWeekNumber() {
    const now = new Date();
    const start = new Date(now.getFullYear(), 0, 1);
    const diff = now - start;
    return Math.ceil((diff / 86400000 + start.getDay() + 1) / 7);
}

// ID 生成
function generateId() {
    return Date.now().toString(36) + Math.random().toString(36).substr(2, 5);
}

// 等待
function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

// 文件下载
function downloadBlob(blob, filename) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    a.click();
    URL.revokeObjectURL(url);
}

// Markdown 简易渲染（用于总结报告）
function renderMarkdown(md) {
    return md
        .replace(/^### (.*$)/gm, '<h3>$1</h3>')
        .replace(/^## (.*$)/gm, '<h2>$1</h2>')
        .replace(/^\- (.*$)/gm, '<li>$1</li>')
        .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
        .replace(/\n/g, '<br>');
}
```
